// ===================================================================== // 🔮 THE STUDY ALCHEMIST CENTRALIZED COMMAND SYSTEM — V3.0 // ===================================================================== const GEMINI_API_KEY = PropertiesService.getScriptProperties().getProperty("GEMINI_API_KEY"); const OUTPUT_FOLDER_ID = PropertiesService.getScriptProperties().getProperty("OUTPUT_FOLDER_ID"); function runAlchemistBitbucketSandbox() { const outputFolder = DriveApp.getFolderById(OUTPUT_FOLDER_ID); // ======================================================================= // 🛰️ STEP 1: DEPLOY CENTRALIZED ROOT ASSETS FROM APPS SCRIPT TO BITBUCKET // ======================================================================= Logger.log("🏁 Initializing Global Root Asset Synchronization..."); const localCSS = HtmlService.createHtmlOutputFromFile('Style').getContent(); const localIndex = HtmlService.createHtmlOutputFromFile('Index').getContent(); pushFileToBitbucket("style.css", localCSS); pushFileToBitbucket("index.html", localIndex); Logger.log("✅ Root infrastructure assets updated successfully."); // Load the prompt schema directly from file asset memory const PROMPT = HtmlService.createHtmlOutputFromFile('Prompt').getContent(); // ======================================================================= // 🔬 STEP 2: RUN THE MULTI-DIMENSIONAL TEXTBOOK PROCESSING MATRIX // ======================================================================= const TEST_MATRIX = [ { "code": "jesc101" }, { "code": "jesc102" }, // Science Ch 1 & 2 (Class 10) { "code": "iesc101" }, { "code": "iesc102" }, // Science Ch 1 & 2 (Class 9) { "code": "jemh101" }, { "code": "jemh102" }, // Math Ch 1 & 2 (Class 10) { "code": "iemh101" }, { "code": "iemh102" } // Math Ch 1 & 2 (Class 9) ]; let count = 0; for (let i = 0; i < TEST_MATRIX.length; i++) { const targetFileCode = TEST_MATRIX[i].code; const ncertUrl = "https://ncert.nic.in/textbook/pdf/" + targetFileCode + ".pdf"; Logger.log("========================================="); Logger.log("🌐 Streaming Online Asset [" + (i+1) + "/" + TEST_MATRIX.length + "]: " + ncertUrl); try { const fetchOptions = { "muteHttpExceptions": true }; const ncertResponse = UrlFetchApp.fetch(ncertUrl, fetchOptions); if (ncertResponse.getResponseCode() !== 200) { Logger.log("❌ ENDPOINT OFFLINE: NCERT HTTP Code " + ncertResponse.getResponseCode() + " for " + targetFileCode); continue; } const pdfBlob = ncertResponse.getBlob().setName(targetFileCode + ".pdf"); Logger.log("📝 Extracting layout text strings via internal Drive engine..."); const pdfText = extractTextFromPdf(pdfBlob); if (!pdfText || pdfText.trim().length === 0) { Logger.log("⚠️ SKIPPED: Extracted dataset returned empty contents."); continue; } const url = "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=" + GEMINI_API_KEY; const payload = { "contents": [{ "parts": [{ "text": PROMPT }, { "text": "Here is the textbook text:\n" + pdfText }] }], "generationConfig": { "responseMimeType": "application/json" } }; const options = { "method": "POST", "contentType": "application/json", "payload": JSON.stringify(payload), "muteHttpExceptions": true }; let response, responseCode, responseText, json, rawAIOutput; let apiSuccess = false; let attempts = 0; const maxRetries = 3; while (attempts < maxRetries && !apiSuccess) { attempts++; response = UrlFetchApp.fetch(url, options); responseCode = response.getResponseCode(); responseText = response.getContentText(); if (responseCode === 200) { json = JSON.parse(responseText); rawAIOutput = json?.candidates?.[0]?.content?.parts?.[0]?.text; if (rawAIOutput) { apiSuccess = true; } } else if (responseCode === 503) { Logger.log("⚠️ SERVER CAPACITY SATURATED (503). Retrying in 15 seconds... (Attempt " + attempts + ")"); Utilities.sleep(15000); } else { Logger.log("❌ Hard Engine Fault: " + responseCode); break; } } if (!apiSuccess) { Logger.log("❌ DROPPED RUN FOR " + targetFileCode); continue; } const aiData = JSON.parse(rawAIOutput); const rawSubject = aiData.subject || "unsorted"; const rawClassLevel = aiData.class_level || "class-unknown"; const rawChapterTitle = aiData.chapter_title || targetFileCode; const htmlContent = aiData.html; const subjectSlug = cleanSlug(rawSubject); const classSlug = cleanSlug(rawClassLevel); const chapterSlug = cleanSlug(rawChapterTitle); const targetBitbucketPath = subjectSlug + "/" + classSlug + "/" + chapterSlug + ".html"; Logger.log("🎯 Compiled Target Tree Path -> " + targetBitbucketPath); outputFolder.createFile(chapterSlug + ".html", htmlContent, MimeType.HTML); pushFileToBitbucket(targetBitbucketPath, htmlContent); count++; Logger.log("⏳ Halting runtime operations for 5 seconds to secure API parameters..."); Utilities.sleep(5000); } catch (e) { Logger.log("❌ Fatal Thread Crash on " + targetFileCode + ": " + e.toString()); } } Logger.log("========================================="); Logger.log("🎉 SUCCESS: Run terminated. Sync completed across all data levels!"); } function cleanSlug(text) { return text.toString().toLowerCase().trim() .replace(/&/g, 'and') .replace(/\s+/g, '-') .replace(/[^\w\-]+/g, '') .replace(/\-\-+/g, '-'); } function extractTextFromPdf(pdfBlob) { const fileResource = { title: pdfBlob.getName(), mimeType: pdfBlob.getContentType() }; const tempDoc = Drive.Files.insert(fileResource, pdfBlob, {convert: true}); const doc = DocumentApp.openById(tempDoc.id); const text = doc.getBody().getText(); Drive.Files.remove(tempDoc.id); return text; } function pushFileToBitbucket(path, content) { const user = PropertiesService.getScriptProperties().getProperty("BITBUCKET_USER"); const repo = PropertiesService.getScriptProperties().getProperty("BITBUCKET_REPO"); const appPassword = PropertiesService.getScriptProperties().getProperty("BITBUCKET_APP_PASS"); const url = "https://api.bitbucket.org/2.0/repositories/" + user + "/" + repo + "/src"; const base64Credential = Utilities.base64Encode(user + ":" + appPassword); const headers = { "Authorization": "Basic " + base64Credential }; const payload = { "message": "🔮 Central Architecture Sync: " + path }; payload[path] = content; const options = { "method": "POST", "headers": headers, "payload": payload, "muteHttpExceptions": true }; const response = UrlFetchApp.fetch(url, options); const responseCode = response.getResponseCode(); if (responseCode === 200 || responseCode === 201) { Logger.log("🚀 LIVE ON REPOSITORY: " + path); return true; } else { Logger.log("❌ REPOSITORY API DEPLOY REJECTION: " + response.getContentText()); return false; } }
