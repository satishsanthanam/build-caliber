const fs = require('fs'); const path = require('path'); // Target paths inside your repository to scan const targetDirectories = ['mathematics', 'science']; function updateHtmlFiles(dirPath) { if (!fs.existsSync(dirPath)) return; const files = fs.readdirSync(dirPath); files.forEach(file => { const fullPath = path.join(dirPath, file); const stat = fs.statSync(fullPath); if (stat.isDirectory()) { updateHtmlFiles(fullPath); // Recursive crawl } else if (path.extname(file).toLowerCase() === '.html' && file !== 'index.html') { let content = fs.readFileSync(fullPath, 'utf8'); // 🛡️ STRICT BYPASS GUARD: Skip files that have already been completely migrated if (content.includes('')) { console.log(`⏭️ Field Verified (Skipped): ${path.relative(__dirname, fullPath)}`); return; } // 🔍 CORE CONTENT EXTRACTION // Dynamically isolates the existing chapter title and inner accordion nodes const titleMatch = content.match(/

([^<]+)<\/h1>/i); const extractedTitle = titleMatch ? titleMatch[1] : "Textbook Chapter"; const containerStart = content.indexOf('

'); const containerEnd = content.lastIndexOf(''); if (containerStart === -1 || containerEnd === -1) { console.log(`⚠️ Skipping non-standard content profile: ${path.relative(__dirname, fullPath)}`); return; } // Isolates raw details boxes, lists, and math elements from the old file const coreHtmlContent = content.substring(containerStart + 5, containerEnd).trim(); // 🏗️ PRISTINE SHELL RE-CONSTRUCTION (Matches Class 6 layout standards exactly) const standardizedHtml = ` ${extractedTitle.replace(/^Chapter\s+\d+:\s*/i, '')}
← Back 🏠 Home

${extractedTitle}

${coreHtmlContent}
`; // Overwrites old framework structure with the pristine mobile configuration format fs.writeFileSync(fullPath, standardizedHtml, 'utf8'); console.log(`♻️ Re-templated & standardized layout: ${path.relative(__dirname, fullPath)}`); } }); } console.log("🏁 Commencing local repository standardization and migration pass..."); targetDirectories.forEach(dir => updateHtmlFiles(path.join(__dirname, dir))); console.log("✅ Standardization pass completed!");
